    import os
import time
import requests
import pandas as pd

API_KEY = os.getenv("OPTIC_ODDS_API_KEY", "cf67bd2c-a2b1-475e-a21a-1ee8623294d5")
BASE = "https://api.opticodds.com/api/v3"
HDRS = {"accept": "application/json", "X-Api-Key": API_KEY}

# 1) Get ACTIVE MLB fixtures after a given timestamp
def get_active_mlb_fixtures(start_date_after_iso: str) -> pd.DataFrame:
    params = {
        "sport": "baseball",
        "league": "mlb",
        "start_date_after": start_date_after_iso,
    }
    r = requests.get(f"{BASE}/fixtures/active", headers=HDRS, params=params, timeout=30)
    r.raise_for_status()
    data = r.json().get("data", [])
    if not data:
        return pd.DataFrame(columns=["id","game_id","start_date","home_team","away_team","status","is_live"])
    rows = []
    for fx in data:
        rows.append({
            "id": fx.get("id"),
            "game_id": fx.get("game_id"),
            "start_date": fx.get("start_date"),
            "home_team": fx.get("home_team_display"),
            "away_team": fx.get("away_team_display"),
            "status": fx.get("status"),
            "is_live": fx.get("is_live"),
        })
    df = pd.DataFrame(rows)
    df["start_date"] = pd.to_datetime(df["start_date"], utc=True, errors="coerce")
    df["date"] = df["start_date"].dt.date
    return df

# 2) (Optional) Pull Moneyline odds for those fixtures
def get_moneyline_for_fixtures(fixture_ids, sportsbooks=None, is_main=True) -> pd.DataFrame:
    if sportsbooks is None:
        sportsbooks = ["BetMGM", "DraftKings", "FanDuel", "Caesars", "Pinnacle"]  # API supports up to 5 per call
    out = []
    for fid in fixture_ids:
        params = {
            "fixture_id": fid,
            "market": "Moneyline",
            "sportsbook": sportsbooks,
        }
        if is_main is not None:
            params["is_main"] = str(is_main).lower()
        r = requests.get(f"{BASE}/fixtures/odds", headers=HDRS, params=params, timeout=30)
        r.raise_for_status()
        for payload in r.json().get("data", []):
            base = {
                "fixture_id": payload.get("id"),
                "game_id": payload.get("game_id"),
                "start_date": payload.get("start_date"),
                "home_team": payload.get("home_team_display"),
                "away_team": payload.get("away_team_display"),
            }
            for o in payload.get("odds", []):
                if (o.get("market_id") or "").lower() != "moneyline":
                    continue
                # prefer explicit price; fall back to OLV/CLV if needed
                price = o.get("price")
                if price is None:
                    price = (o.get("olv") or {}).get("price")
                if price is None:
                    price = (o.get("clv") or {}).get("price")
                out.append({
                    **base,
                    "sportsbook": o.get("sportsbook"),
                    "market": o.get("market"),
                    "name": o.get("name"),  # team for ML
                    "is_main": o.get("is_main"),
                    "american_odds": pd.to_numeric(price, errors="coerce"),
                })
        time.sleep(0.15)  # be polite
    df = pd.DataFrame(out)
    if df.empty:
        return df
    df["start_date"] = pd.to_datetime(df["start_date"], utc=True, errors="coerce")
    df["date"] = df["start_date"].dt.date
    return df

def _norm_code(raw: str) -> Optional[str]:
    if not raw:
        return None
    x = raw.strip().upper()
    return TEAM_MAP.get(x, x)

MON = {"JAN":1,"FEB":2,"MAR":3,"APR":4,"MAY":5,"JUN":6,"JUL":7,"AUG":8,"SEP":9,"OCT":10,"NOV":11,"DEC":12}

def _parse_ticker_teams_and_date(ticker: str, title: Optional[str]) -> Dict[str, Any]:
    out = {"date": None, "away": None, "home": None}
    t = (ticker or "").upper()

    # KXMLBGAME-YYMMMDD<AWAY><HOME>[-<SIDE>]
    m = re.search(r"KXMLBGAME-(\d{2})([A-Z]{3})(\d{2})([A-Z]{2,3})([A-Z]{2,3})", t)
    if m:
        yy, mon, dd, away3, home3 = m.groups()
        year  = 2000 + int(yy)
        month = MON.get(mon)
        if month:
            out["date"] = datetime(year, month, int(dd), tzinfo=timezone.utc).date()
        out["away"] = _norm_code(away3)
        out["home"] = _norm_code(home3)

    return out

def _pick_prob(row: Dict[str, Any]) -> Optional[float]:
    """
    Pick a probability from Kalshi market prices.
    Preference: last_price -> mid(yes_bid,yes_ask) -> yes_ask -> yes_bid.
    Convert cents to [0,1].
    """
    for key in ("last_price",):
        v = row.get(key)
        if isinstance(v, (int, float)):
            return v / 100.0
    yb, ya = row.get("yes_bid"), row.get("yes_ask")
    if isinstance(yb, (int, float)) and isinstance(ya, (int, float)) and yb >= 0 and ya > 0:
        return ((yb + ya) / 2.0) / 100.0
    if isinstance(ya, (int, float)) and ya > 0:
        return ya / 100.0
    if isinstance(yb, (int, float)) and yb >= 0:
        return yb / 100.0
    return None

def kalshi_get_active_mlb_markets(start_date_after_iso: str,
                                  end_date_iso: Optional[str] = None,
                                  limit_per_page: int = 1000,
                                  max_pages: int = 20) -> pd.DataFrame:
    """
    Pull active MLB game markets (KXMLBGAME) that close after start_date_after_iso.
    Optionally cap with end_date_iso.
    """
    def to_epoch(s: str) -> int:
        return int(pd.to_datetime(s, utc=True).timestamp())

    params = {
        "series_ticker": "KXMLBGAME",  # MLB game series
        "status": "unopened,open",     # active
        "min_close_ts": to_epoch(start_date_after_iso),
        "limit": limit_per_page,
    }
    if end_date_iso:
        params["max_close_ts"] = to_epoch(end_date_iso)

    markets: List[Dict[str, Any]] = []
    cursor = None
    pages = 0

    while True:
        q = dict(params)
        if cursor:
            q["cursor"] = cursor
        r = SESSION.get(f"{KALSHI_BASE}/markets", params=q, timeout=30)
        r.raise_for_status()
        js = r.json()
        mkts = js.get("markets", []) or []
        markets.extend(mkts)
        cursor = js.get("cursor")
        pages += 1
        if not cursor or pages >= max_pages:
            break
        time.sleep(0.1)

    # Flatten to rows (one row per market/side)
    rows = []
    for m in markets:
        ticker = m.get("ticker")
        title = m.get("title")
        parsed = _parse_ticker_teams_and_date(ticker, title)

        row = {
            "ticker": ticker,
            "title": title,
            "subtitle": m.get("subtitle"),
            "status": m.get("status"),
            "close_time": pd.to_datetime(m.get("close_time"), utc=True, errors="coerce"),
            "away": parsed["away"],
            "home": parsed["home"],
            "date": parsed["date"],
            "yes_bid": m.get("yes_bid"),
            "yes_ask": m.get("yes_ask"),
            "last_price": m.get("last_price"),
            "kalshi_prob": _pick_prob(m),  # 0..1
            "yes_sub_title": m.get("yes_sub_title"),
            "no_sub_title": m.get("no_sub_title"),
        }
        rows.append(row)

    df = pd.DataFrame(rows)
    if not df.empty:
        # ensure types
        for c in ("yes_bid", "yes_ask", "last_price"):
            if c in df.columns:
                df[c] = pd.to_numeric(df[c], errors="coerce")
        # coerce date (if we only got close_time)
        if "date" in df.columns and df["date"].isna().all() and "close_time" in df.columns:
            df["date"] = df["close_time"].dt.date
    return df

    import re
import pandas as pd
import numpy as np

# --- team maps ---
TEAM_TO_CODE = {
    "Baltimore Orioles":"BAL","Boston Red Sox":"BOS","New York Yankees":"NYY","Tampa Bay Rays":"TB","Toronto Blue Jays":"TOR",
    "Chicago White Sox":"CWS","Cleveland Guardians":"CLE","Detroit Tigers":"DET","Kansas City Royals":"KC","Minnesota Twins":"MIN",
    "Houston Astros":"HOU","Los Angeles Angels":"LAA","Oakland Athletics":"OAK","Seattle Mariners":"SEA","Texas Rangers":"TEX",
    "Atlanta Braves":"ATL","Miami Marlins":"MIA","New York Mets":"NYM","Philadelphia Phillies":"PHI","Washington Nationals":"WSH",
    "Chicago Cubs":"CHC","Cincinnati Reds":"CIN","Milwaukee Brewers":"MIL","Pittsburgh Pirates":"PIT","St. Louis Cardinals":"STL",
    "Arizona Diamondbacks":"AZ","Colorado Rockies":"COL","Los Angeles Dodgers":"LAD","San Diego Padres":"SD","San Francisco Giants":"SF",
}
KALSHI_CODE_FIX = {"ATH":"OAK","KCR":"KC","TBR":"TB","WAS":"WSH","SDP":"SD","SDN":"SD","YM":"NYM","SFG":"SF","ARI":"AZ"}
def fix_code(x):
    if x is None: return None
    return KALSHI_CODE_FIX.get(str(x).strip().upper(), str(x).strip().upper())

def amer_to_prob(odds):
    if pd.isna(odds): return np.nan
    o = float(odds)
    return 100.0/(o+100.0) if o > 0 else (-o)/((-o)+100.0)

def parse_kalshi_side_from_ticker(t):
    m = re.search(r"-([A-Z]{2,3})$", str(t).upper())
    return fix_code(m.group(1)) if m else None

# ---------- OPTIC: build per-side consensus (safe against duplicates) ----------
def build_optic_consensus(optic_moneyline_df: pd.DataFrame) -> pd.DataFrame:
    df = optic_moneyline_df.copy()

    # ensure datetime + date
    if "start_date" in df.columns:
        df["start_date"] = pd.to_datetime(df["start_date"], utc=True, errors="coerce")
        df["date"] = df["start_date"].dt.date
    else:
        df["date"] = pd.to_datetime(df["date"], utc=True, errors="coerce").dt.date

    # normalize teams & side to codes
    df["away_code"] = df["away_team"].map(TEAM_TO_CODE)
    df["home_code"] = df["home_team"].map(TEAM_TO_CODE)
    df["side_code"] = df["name"].map(TEAM_TO_CODE)

    # moneyline only; optional: only main lines
    df = df[df["market"].str.contains("moneyline", case=False, na=False)]
    if "is_main" in df.columns:
        df = df[df["is_main"].fillna(True)]

    # implied prob
    df["sb_prob_raw"] = df["american_odds"].apply(amer_to_prob)

    # collapse to ONE row per (date, game_id, away, home, sportsbook, side)
    base = ["date","game_id","away_code","home_code","sportsbook","side_code"]
    df2 = (df[base + ["sb_prob_raw"]]
           .dropna(subset=["away_code","home_code","side_code","sb_prob_raw"])
           .groupby(base, as_index=False)
           .agg(sb_prob_raw=("sb_prob_raw","mean")))

    # pivot per game & book (safe: pivot_table with mean)
    probs = (pd.pivot_table(df2,
                            index=["date","game_id","away_code","home_code","sportsbook"],
                            columns="side_code",
                            values="sb_prob_raw",
                            aggfunc="mean"))
    # de-vig row-wise if we have both sides
    s = probs.sum(axis=1)
    probs_norm = probs.div(s, axis=0)

    # back to long
    stacked = (probs_norm.stack().rename("sb_prob").reset_index()
               .rename(columns={"side_code":"side_code"}))

    # consensus across books; also track #books and #games contributing
    optic_consensus = (stacked.groupby(["date","away_code","home_code","side_code"], as_index=False)
                       .agg(optic_prob_mean=("sb_prob","mean"),
                            optic_prob_med=("sb_prob","median"),
                            books=("sportsbook","nunique"),
                            games=("game_id","nunique")))
    return optic_consensus

# ---------- KALSHI: normalize per side ----------
def build_kalshi_sides(kalshi_df: pd.DataFrame) -> pd.DataFrame:
    k = kalshi_df.copy()
    k["date"] = pd.to_datetime(k["date"], utc=True, errors="coerce").dt.date
    k["away_code"] = k["away"].map(fix_code)
    k["home_code"] = k["home"].map(fix_code)
    k["side_code"] = k["ticker"].apply(parse_kalshi_side_from_ticker)

    k = k.dropna(subset=["date","away_code","home_code","side_code","kalshi_prob"])
    # keep last/avg per market side if duplicates
    k = (k.groupby(["date","away_code","home_code","side_code"], as_index=False)
           .agg(kalshi_prob=("kalshi_prob","mean"),
                ticker=("ticker","last"),
                title=("title","last")))
    return k

# ---------- final merge ----------
def make_comparison_df(optic_moneyline_df: pd.DataFrame, kalshi_df: pd.DataFrame) -> pd.DataFrame:
    o = build_optic_consensus(optic_moneyline_df)
    k = build_kalshi_sides(kalshi_df)

    merged = k.merge(o, on=["date","away_code","home_code","side_code"], how="inner")
    merged["delta"] = merged["kalshi_prob"] - merged["optic_prob_mean"]

    merged = (merged.groupby(["date","away_code","home_code","side_code"], as_index=False)
                    .agg(kalshi_prob=("kalshi_prob","mean"),
                         optic_prob_mean=("optic_prob_mean","mean"),
                         optic_prob_med=("optic_prob_med","median"),
                         books=("books","max"),
                         games=("games","max"),
                         example_ticker=("ticker","last"),
                         example_title=("title","last")))
    merged["delta"] = merged["kalshi_prob"] - merged["optic_prob_mean"]

    cols = ["date","away_code","home_code","side_code",
            "kalshi_prob","optic_prob_mean","optic_prob_med","books","games","delta",
            "example_ticker","example_title"]
    return merged[cols].sort_values(["date","away_code","home_code","side_code"]).reset_index(drop=True)
